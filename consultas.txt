1    SELECT pp.*, p.* FROM planes_participantes pp, planes p where pp.cod_plan=p.cod_plan



2    SELECT pr.*, pp.*, p.* FROM planes_participantes pp, planes p, participantes pr where pp.cod_plan=p.cod_plan and pp.cod_part=pr.cod_par



3   SELECT p.*, (SELECT COUNT(cod_part) from planes_participantes where cod_plan=p.cod_plan) as n_inscritos FROM planes p


4   SELECT pr.*, p.*, r.* FROM planes_responsables pr, planes p, responsables r where pr.cod_plan=p.cod_plan
 and pr.cod_resp=r.cod_resp


5  SELECT r.ced_resp, r.nomb_resp, r.apell_resp, r.espe_resp FROM `responsables` r, `nivel_academico` n WHERE n.id_nivel=r.id_nivel


6 SELECT s.*, p.cedu_pac, p.nomb_pac, p.diag_pac, ts.nomb_tipo_serv  FROM `servicios` s, `paciente` p, `tipo_servicio` ts WHERE ts.id_tipo_serv=s.id_tipo_serv

7  select count(pp.cod_part) as cantidad_participantes, pp.cod_plan, (select nomb_plan from planes where cod_plan=pp.cod_plan) as nombre_plan from planes_participantes pp group by pp.cod_plan

8  select count(pp.cod_part) as cantidad_participantes, pp.cod_plan, (select nomb_plan from planes where cod_plan=pp.cod_plan) as nombre_plan, (select tipo_plan from planes where cod_plan=pp.cod_plan) as tipo_plan from planes_participantes pp group by pp.cod_plan order by tipo_plan ASC

9  SELECT *  FROM planes WHERE cod_plan = '9'

10  select count(pp.cod_part) as cantidad_participantes, pp.cod_plan, (select nomb_plan from planes where cod_plan=pp.cod_plan) as nombre_plan, (select cant_unid from planes where cod_plan=pp.cod_plan) as cantidad_unidad, (select id_unid from planes where cod_plan=pp.cod_plan) as codigo_unid, (select valor_unid from unidades_trib where id_unid=codigo_unid) as valor_unidad from planes_participantes pp group by pp.cod_plan

11  select count(pp.cod_part) as cantidad_participantes, pp.cod_plan, (select nomb_plan from planes where cod_plan=pp.cod_plan) as nombre_plan, (select cant_unid from planes where cod_plan=pp.cod_plan) as cantidad_unidad, (select id_unid from planes where cod_plan=pp.cod_plan) as codigo_unid, (select valor_unid from unidades_trib where id_unid=codigo_unid) as valor_unidad, ((count(pp.cod_part))*(select cant_unid from planes where cod_plan=pp.cod_plan)*(select valor_unid from unidades_trib where id_unid=codigo_unid)) as total_curso from planes_participantes pp group by pp.cod_plan
